function checkExportFormat(limit) {

    var msg = "";

    for (i=0;i&lt;document.exportCite.elements.length;i++) {
        if (document.exportCite.elements[i].value == "REFWX") {
            if (document.exportCite.elements[i].checked) {
                document.exportCite.setAttribute('target',"RefWorksMain");
                if (checkLimit()) {

                    }

                    alert(msg);
                }

                document.exportCite.setAttribute('target',"");

            break;
        }
    }
    return true;
}

function determineCount(numStr) {
    var count=1;

    if(numStr.indexOf("-")) {
        var rangeStr = numStr.split("-");

        if(rangeStr.length == 2) {
            var highNum = parseInt(rangeStr[1]);
            var lowNum = parseInt(rangeStr[0]);

            if(highNum &gt;= lowNum) { count = (highNum - lowNum) + 1; }
        }
    }
    return count;
}

function checkLimit(){

    var selectNum = "";
    var useSelected = false;
    var allNum = 0;
    var count = 0;
    var rdoc = 0;

    for (i=0;i&lt;document.exportCite.elements.length;i++) {
        if (document.exportCite.elements[i].name == "count") {
            allNum = document.exportCite.elements[i].value;
        }
        if (document.exportCite.elements[i].name == "NUMBER_LIST") {
            selectNum = document.exportCite.elements[i].value;
        }
        if (document.exportCite.elements[i].name == "_rdoc") {
            rdoc = document.exportCite.elements[i].value;
        }
        if (document.exportCite.elements[i].value == "selected") {
            if (document.exportCite.elements[i].checked) {
                useSelected = true;
            }
        }
    }

    if (useSelected) {
        selectNum = selectNum.replace(/ /g,"");
        if(selectNum.indexOf(",")) {
            var commaList = selectNum.split(",");
            for(i=0; i &lt; commaList.length; i++) {
                count += determineCount(commaList[i]);
            }
        }
        else {
            count += determineCount(selectNum);
        }
    }
    else { 
        if (rdoc &gt; 0) {
            count = 1;
        }
        else {
            count = allNum;
        }
    }

    if(count &gt; 50) { return true; }
  
    return false;
}

</ul></li></ul></dd></dl></div><!--FRAGMENTEND--><div class="page_fragment_ind auth_frag" data-id="frag_2"></div></div><div id="frag_2" class="page_fragment" data-fid="2"><script>prs.rt("abs_1st_beg");</script><!--VALIDHTML--><hr id="abs_author1" class="artHeader" /><div class="abstract svAbstract " data-etype="ab"><h2 class="secHeading" id="author1">Abstract</h2><p id="">A meta-analysis was carried out, in order to summarise published data on the relationship between breast cancer, fruit and vegetable consumption and/or the intake of beta-carotene and vitamin C. Relative risks were extracted from 26 published studies from 1982 to 1997. Random and fixed effects models were used. Between studies, heterogeneity was found for vegetables, fruit, vitamin C but not for beta-carotene. Summary relative risk (RR) estimates based upon a random effects model, except for beta-carotene, for ‘high consumption’ compared with ‘low consumption’, derived from the studies satisfying the inclusion criteria were as follows: vegetable consumption: RR=0.75 (95% CI (confidence interval) 0.66–0.85) from 17 studies; fruit consumption: RR=0.94 (95% CI 0.79–1.11) from 12 studies; vitamin C: RR=0.80 (95% CI 0.68–0.95) from 9 studies; beta-carotene: RR=0.82 (95% CI 0.76–0.91) from 11 studies. This analysis confirms the association between intake of vegetables and, to a lesser extent, fruits and breast cancer risk from published sources. Increasing vegetable consumption might reduce the risk of breast cancer.</p></div><!--VALIDHTML--><div id="embeddedPDF" style="display:none"></div><script>prs.rt("abs_end");</script><div id="SD_BA1P" class="sgfNoTitleBar sgfNoGadgetBorder svDoNotLink ui-sortable"></div><!--VALIDHTML--><h2 id="kwd_1" class="svKeywords">Keywords</h2><ul class="keyword" id=""><li id="" class="svKeywords"><span id="">Meta-analysis</span>; </li><li id="" class="svKeywords"><span id="">Breast</span>; </li><li id="" class="svKeywords"><span id="">Cancer</span>; </li><li id="" class="svKeywords"><span id="">Vegetables</span>; </li><li id="" class="svKeywords"><span id="">Fruit</span>; </li><li id="" class="svKeywords"><span id="">Vitamin C</span>; </li><li id="" class="svKeywords"><span id="">Beta-carotene</span></li></ul><!--VALIDHTML--><hr class="artHeader" /><h2 class="svArticle" id="section_lbl1">1. Introduction</h2><p class="svArticle section" id="">Breast cancer is a major public health problem in industrialised countries. It is estimated that one in eight American women will develop breast cancer in her lifetime <a href="#BIB1" id="bBIB1" class="intra_ref">1</a> and <a href="#BIB2" id="bBIB2" class="intra_ref">2</a>. The most convincing evidence that breast cancer is possibly preventable is that the rate of breast cancer increases after a woman migrates from a country with a low rate to one with a higher rate. Dietary change has been consistently suggested as a major contributor factor to this situation <a href="#BIB3" id="bBIB3" class="intra_ref">3</a>, <a href="#BIB4" id="bBIB4" class="intra_ref">4</a>, <a href="#BIB5" id="bBIB5" class="intra_ref">5</a> and <a href="#BIB6" id="bBIB6" class="intra_ref">6</a>.</p><!--VALIDHTML--><p class="svArticle section" id="">When breast cancer risk has been examined in relation to intake of vegetables, fruits or both as a food group the results provide some evidence of a protective effect <span id="bBIB7"><a href="#BIB7" id="ancbBIB7" class="intra_ref">[7]</a></span>. Fruits and vegetables are common sources of many candidate protective substances, including carotenoids, with and without vitamin A activity, and ascorbic acid. Carotenoids and vitamin C may protect against breast cancer due to their role in antioxidant defence.</p><!--VALIDHTML--><p class="svArticle section" id="">Amongst epidemiological studies, there is a convincing pattern showing that women with diets which are high in vegetables and fruits have a decreased risk of breast cancer. The aim of this work was to analyse published results that explore the relationship between breast cancer risk and the consumption of fruits and vegetables and two associated micronutrients (vitamin C and beta-carotene).
